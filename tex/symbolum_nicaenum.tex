\input{preambulum.tex}
\usepackage[margin=1in]{geometry}
\usepackage{parskip}

\begin{document}
    % I BELIEVE in one God,
    \lettrine{C}{redo} in unum Deum,
    % the Father almighty,
    Patrem omnipotentem,
    % maker of heaven and earth,
    factorem caeli et terrae,
    % of all things visible and invisible.
    visibilium omnium et invisibilium.

    % I believe in one Lord Jesus Christ,
    \lettrine{E}{t} in unum Dominum Iesum Christum,
    % the only begotten Son of God,
    Filium Dei unigenitum,
    % born of the Father before all ages.
    et ex Patre natum ante omnia saecula.
    
    % God from God, Light from Light, true God from true God,
    Deum de Deo, Lumen de Lumine, Deum verum de Deo vero, 
    % begotten, not made, consubstantial with the Father;
    genitum, non factum, consubstantialem Patri;
    % through him all things were made.
    per quem omnia facta sunt.
    
    % For us men and for our salvation
    Qui propter nos homines et propter nostram salutem
    % he came down from heaven,
    descendit de caelis.
    % and by the power of the Holy Spirit was incarnate of the Virgin Mary, and became man.
    Et incarnatus est de Spiritu Sancto ex Maria Virgine, et homo factus est.
    
    % For our sake he was crucified under Pontius Pilate,
    Crucifixus etiam pro nobis sub Pontio Pilato,
    % he suffered death and was buried,
    passus et sepultus est,
    % and rose again on the third day in accordance with the Scriptures.
    et resurrexit tertia die, secundum Scripturas,
    % He ascended into heaven, and is seated at the right hand of the Father.
    et ascendit in caelum, sedet ad dexteram Patris.
    
    % He will come again in glory
    Et iterum venturus est cum gloria,
    % to judge the living and the dead,
    iudicare vivos et mortuos,
    % and his kingdom will have no end.
    cuius regni non erit finis.

    % I believe in the Holy Spirit, the Lord the giver of life,
    \lettrine{E}{t} in Spiritum Sanctum, Dominum et vivificantem,
    % who proceeds from the Father and the Son,
    qui ex Patre Filioque procedit.
    % who with the Father and the Son is adored and glorified,
    Qui cum Patre et Filio simul adoratur et conglorificatur;
    % who has spoken through the prophets.
    qui locutus est per prophetas.

    % I believe in one, holy, catholic, and apostolic Church.
    \lettrine{E}{t} unam, sanctam, catholicam et apostolicam Ecclesiam.
    % I confess one baptism for the forgiveness of sins,
    Confiteor unum baptisma in remissionem peccatorum.
    % and I look forward to the resurrection of the dead
    Et expecto resurrectionem mortuorum,
    % and the life of the world to come.
    et vitam venturi saeculi.
    % Amen.
    \amen.
\end{document}
