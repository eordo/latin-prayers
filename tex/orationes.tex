\begin{document}
    \begin{center}
        % In the name of the Father, the Son, and the Holy Spirit. Amen.
        \signumcrucis
    \end{center}

    % OUR FATHER, who art in heaven,
    \lettrine[lines=3]{P}{ater noster,} qui es in caelis,
    % hallowed be thy name.
    sanctificetur nomen tuum.
    % Thy kingdom come.
    Adveniat regnam tuum.
    % Thy will be done on earth as it is in heaven.
    Fiat voluntas tua sicut in caelo et in terra.
    % Give us this day our daily bread,
    Panem nostrum quotidianum da nobis hodie,
    % and forgive us our trespasses
    et dimitte nobis debita nostra
    % as we forgive those who trespass against us.
    sicut et nos dimittimus debitoribus nostris.
    % And lead us not into temptation, but deliver us from evil.
    Et ne nos inducas in tentationem, sed libera nos a malo.
    % Amen.
    \amen.

    \begin{center}
        \pgfornament[color=Maroon,width=0.125\textwidth]{70}
    \end{center}

    % HAIL MARY, full of grace,
    \lettrine[lines=3]{A}{ve Maria,} gratia plena,
    % the Lord is with thee.
    Dominus tecum.
    % Blessed art thou amongst women,
    Benedicta tu in mulieribus,
    % and blessed is the fruit of thy womb, Jesus.
    et benedictus fructus ventris tui, Iesus.
    % Holy Mary, Mother of God,
    Sancta Maria, Mater Dei,
    % pray for us sinners,
    ora pro nobis peccatoribus,
    % now and at the hour of our death.
    nunc et in hora mortis nostrae.
    % Amen.
    \amen.

    \begin{center}
        \pgfornament[color=Maroon,width=0.125\textwidth]{70}
    \end{center}

    % GLORY BE TO THE FATHER, and to the Son, and to the Holy Spirit,
    \lettrine[lines=3]{G}{loria Patri}, et Filio, et Spiritui Sancto,
    % as it was in the beginning, is now, and ever shall be, world without end.
    sicut erat in principio, et nunc, et semper, et in saecula saeculorum.
    % Amen.
    \amen.

    \begin{center}
        % In the name of the Father, the Son, and the Holy Spirit. Amen.
        \signumcrucis
    \end{center}
\end{document}
